
// Firestore 
rules_version="2";
service cloud.firestore{
    match /databases/{databases}/documents {
        //listing
        match /listing/{listing}{
            allow read;
            allow create :if request.auth !=null && request.resource.data.imgUrls.size()<7;
            allow delete: if resource.userRef==request.auth.uid;
            allow update : if resource.userRef==request.auth.uid
        }
        // Users
        match /users/{user}{
            allow read;
            allow create;
            allow update : if request.auth.uid ==user
        }
    }
}

// storage rules

rules_version ="2";
service firebase.storage {
    match /b/{bucket}/o {
        match / {allPaths=**}{
            allow read;
            allow write : if request.auth != null && request.resource.size <2 * 1024 * 1024 && request.resource.contentType.matches("image/.*")
        }
    }
}
